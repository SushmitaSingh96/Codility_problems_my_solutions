sub solution {
    my (@A)=@_;
    # write your code in Perl 5.18
    print "@A";
    sort @A;
    my $tmp = $A[0];
    my $i = 0;
    while ($i< scalar @A){
        if ($i == scalar @A -1){
            return $tmp[$i];
        }
        if ($tmp[$i] == $tmp[$i+1]){
            $i = $i+2;
            next;
        }
        else{
            return $tmp[$i];
        }
    }
}
